-- tables
Create table user(
u_id number primary key Generated by default as Identity,  -- not sure if will be using identity/probabily sequence
email varchar(30) unique,
name varchar(30) not null,
location varchar(30) not null,
mobile_no varchar(10) check(length(mobile_no)=10)
);

create table movie(
m_id number primary key generated by default as identity,
m_name varchar(30),
m_description varchar(60),
m_genre varchar(30),
m_length number,
m_price number,
m_language varchar(30),
m_format varchar(30)
);

create table theater(
t_id number primary key generated by default as identity,
t_name varchar(60),
t_location varchar(30)
);


create table ticket(
ticket_id number primary key generated by default as identity,
u_id number,
m_id number,
ticket_date Date,
ticket_time Timestamp,
ticket_seat varchar(10),
ticket_amount number,
Foreign key(u_id) References user(u_id),
Foreign key(m_id) References movie(m_id)
);
